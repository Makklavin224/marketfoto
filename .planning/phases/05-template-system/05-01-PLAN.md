```python
from pydantic import BaseModel
from uuid import UUID


class TemplateListItem(BaseModel):
    """Template summary for list endpoint (no config). Per D-02."""
    id: UUID
    name: str
    slug: str
    category: str
    preview_url: str
    is_premium: bool
    marketplace: list[str]

    model_config = {"from_attributes": True}


class TemplateDetail(BaseModel):
    """Full template with config JSON. Per D-03."""
    id: UUID
    name: str
    slug: str
    category: str
    preview_url: str
    config: dict
    is_premium: bool
    marketplace: list[str]
    sort_order: int

    model_config = {"from_attributes": True}


class TemplateListResponse(BaseModel):
    """Wrapper response for list endpoint."""
    templates: list[TemplateListItem]


class TemplateDetailResponse(BaseModel):
    """Wrapper response for detail endpoint."""
    template: TemplateDetail
```

**backend/app/api/templates.py** -- Templates router with 2 endpoints:

Create `router = APIRouter(prefix="/api/templates", tags=["templates"])`.

**GET /api/templates** (per D-02, TMPL-01):
- Optional query params: `category: str | None = None`, `marketplace: str | None = None`
- Create async session via `AsyncSessionLocal()`
- Build SQLAlchemy select query on Template model
- If `category` provided: add `.where(Template.category == category)`
- If `marketplace` provided: add `.where(Template.marketplace.any(marketplace))` (PostgreSQL ARRAY `ANY` operator via SQLAlchemy `.any()`)
- Order by `Template.sort_order.asc(), Template.name.asc()`
- Execute query, convert results to `TemplateListItem` using `model_validate`
- Return `TemplateListResponse(templates=[...])`

**GET /api/templates/{template_id}** (per D-03, D-06, TMPL-04):
- Path param: `template_id: UUID`
- Optional header: Authorization (for premium check). Use a soft auth approach: try to extract user from JWT if present, but don't require auth for browsing. For premium gating, if no auth header or user is on free plan and template.is_premium is True, return 403.
- Fetch template by id. If not found, return 404 with `{"error": "Template not found"}`.
- If template.is_premium is True: check if user is authenticated and on a paid plan. If user is not authenticated or user.plan == "free", return 403 with `{"error": "Этот шаблон доступен по подписке. Оформите подписку для доступа.", "upgrade_url": "/pricing"}`.
- Otherwise return `TemplateDetailResponse(template=TemplateDetail.model_validate(template))`.

For the auth dependency, create a lightweight `get_optional_current_user` dependency that:
1. Checks for `Authorization: Bearer <token>` header
2. If present, decodes JWT using `settings.jwt_secret` and `settings.jwt_algorithm` via PyJWT
3. Looks up User by id from token payload
4. Returns User or None (never raises — optional auth)

Import User model from `app.models.user`. Import `settings` from `app.config`. Import `jwt` (PyJWT). Import `AsyncSessionLocal` from `app.database`.

```python
async def seed_templates() -> None:
    """Insert seed templates. Idempotent via ON CONFLICT DO NOTHING on slug."""
    async with AsyncSessionLocal() as session:
        for tmpl_data in SEED_TEMPLATES:
            # Check if template with this slug already exists
            result = await session.execute(
                text("SELECT id FROM templates WHERE slug = :slug"),
                {"slug": tmpl_data["slug"]},
            )
            if result.scalar() is not None:
                print(f"  Template '{tmpl_data['slug']}' already exists, skipping.")
                continue

            template = Template(
                name=tmpl_data["name"],
                slug=tmpl_data["slug"],
                category=tmpl_data["category"],
                is_premium=tmpl_data["is_premium"],
                marketplace=tmpl_data["marketplace"],
                sort_order=tmpl_data["sort_order"],
                preview_url=tmpl_data["preview_url"],
                config=tmpl_data["config"],
            )
            session.add(template)
            print(f"  Seeded template: {tmpl_data['slug']}")

        await session.commit()
    print("Template seeding complete.")


if __name__ == "__main__":
```
