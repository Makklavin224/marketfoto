**File 1: frontend/src/hooks/useImageStatus.ts**

Create a custom hook using @tanstack/react-query's useQuery with polling:

```typescript
export function useImageStatus(imageId: string | null) {
  return useQuery({
    queryKey: ['image-status', imageId],
    queryFn: async () => {
      const { data } = await axios.get(`/api/images/${imageId}/status`);
      return data as ImageStatusData;
    },
    enabled: !!imageId,
    refetchInterval: (query) => {
      // Poll every 2 seconds while processing (per D-05)
      const status = query.state.data?.status;
      if (status === 'processing') return 2000;
      return false; // Stop polling when processed or failed
    },
    // Max 20 polling attempts = 40 seconds (per SPEC line 179)
    // After that, show client-side timeout error
    retry: false,
  });
}
```

Type the response:
```typescript
interface ImageStatusData {
  status: 'uploaded' | 'processing' | 'processed' | 'failed';
  processed_url: string | null;
  error_message: string | null;
  processing_time_ms: number | null;
}
```

Add a polling attempt counter using useRef. After 20 refetches (40 seconds), set a local `timedOut` state to true and stop polling. Return { data, isLoading, timedOut } from the hook.
